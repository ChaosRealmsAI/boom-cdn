Code Block

stratum.yaml 
# Stratum pipeline manifest
version: "2"
pipeline:
  name: realtime-analytics
  region: us-east-1
  sources:
    - type: kafka
      topic: events.raw
      parallelism: 8
  transforms:
    - filter("event_type != 'heartbeat'")
    - enrich(source="geo-db", field="ip_address")
    - aggregate(window="5m", key="user_id")
  sinks:
    - type: clickhouse
      table: analytics.sessions
